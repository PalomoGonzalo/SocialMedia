create procedure [dbo].[spPaginacion](
@pagina int,
@CantidadReg int)
as

begin
DECLARE @SkipReG int = @pagina*@CantidadReg

SELECT * FROM Publicacion
ORDER BY IdPublicacion
OFFSET @SkipReG ROW
FETCH NEXT @CantidadReg ROWS ONLY

end
GO
